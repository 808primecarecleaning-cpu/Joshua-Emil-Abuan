  <script>
    // small helpers: form -> mailto fallback and minimal validation + year
    document.getElementById('year').textContent = new Date().getFullYear();

    function mailtoQuick(){
      window.location.href = 'mailto:hello@primecare.example?subject=Quick%20Question';
    }

    function handleFormSubmit(e){
      e.preventDefault();
      const form = e.target;
      const name = form.name.value.trim();
      const email = form.email.value.trim();
      const phone = form.phone.value.trim();
      const service = form.service.value;
      const notes = form.notes.value.trim();

      if(!name || !email){
        alert('Please provide your name and email.');
        return false;
      }

      // Build mailto - ensures recipients get a copy even if form backend not configured
      const subject = encodeURIComponent('Estimate Request — ' + service + ' — ' + name);
      const body = encodeURIComponent(
        `Name: ${name}\nEmail: ${email}\nPhone: ${phone}\nService: ${service}\n\nNotes/Address:\n${notes}`
      );

      // open user's email client with filled message
      const mailto = `mailto:hello@primecare.example?subject=${subject}&body=${body}`;
      window.location.href = mailto;

      // Optionally show a friendly local confirmation
      alert('Estimate request opened in your email client. If nothing opened, email hello@primecare.example directly.');

      return false;
    }
  </script>
